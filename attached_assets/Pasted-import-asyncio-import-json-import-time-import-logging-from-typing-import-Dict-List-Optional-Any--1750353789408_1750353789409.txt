    batch_size: int = 10
    rate_limit_rpm: int = 60
    cost_per_token: float = 0.0
    quality_score: float = 0.0
    specialized_for: List[TaskType] = None

    def get_optimal_model(self, task_type: TaskType, budget_limit: float = None, 
                         quality_threshold: float = 0.85) -> str:
        """Intelligent model selection based on task type, budget, and quality requirements"""
        
        suitable_models = []
        
        for model_name, config in self.model_configs.items():
            # Check if model is specialized for this task type
            if config.specialized_for and task_type in config.specialized_for:
                # Check quality threshold
                if config.quality_score >= quality_threshold:
                    # Check budget constraint
                    if budget_limit is None or config.cost_per_token <= budget_limit:
                        suitable_models.append((model_name, config))
        
        if not suitable_models:
            # Fallback to any model meeting quality threshold
            suitable_models = [
                (name, config) for name, config in self.model_configs.items()
                if config.quality_score >= quality_threshold
            ]
        
        if not suitable_models:
            # Final fallback to best available model
            suitable_models = [(name, config) for name, config in self.model_configs.items()]
        
        # Sort by quality score (descending) and cost (ascending)
        suitable_models.sort(key=lambda x: (-x[1].quality_score, x[1].cost_per_token))
        
        return suitable_models[0][0] if suitable_models else "gpt-3.5-turbo"

    async def generate_response(self, prompt: str, model_name: str, 
                              task_type: TaskType = None, **kwargs) -> LLMResponse:
        """Enhanced response generation with caching, optimization, and monitoring"""
        
        start_time = time.time()
        
        # Check cache first
        cache_key = self._generate_cache_key(prompt, model_name, kwargs)
        if cache_key in self.response_cache:
            cached_response = self.response_cache[cache_key]
            cached_response.cached = True
            return cached_response
        
        config = self.model_configs.get(model_name)
        if not config:
            raise ValueError(f"Unknown model: {model_name}")
        
        try:
            # Generate response based on provider
            if config.provider == LLMProvider.OPENAI:
                response = await self._generate_openai_response(prompt, config, **kwargs)
            elif config.provider == LLMProvider.ANTHROPIC:
                response = await self._generate_anthropic_response(prompt, config, **kwargs)
            elif config.provider == LLMProvider.GOOGLE:
                response = await self._generate_google_response(prompt, config, **kwargs)
            elif config.provider == LLMProvider.OLLAMA:
                response = await self._generate_ollama_response(prompt, config, **kwargs)
            else:
                raise ValueError(f"Unsupported provider: {config.provider}")
            
            # Calculate metrics
            response_time = time.time() - start_time
            response.response_time = response_time
            response.timestamp = datetime.now()
            
            # Cache response if enabled
            if config.cache_enabled:
                self.response_cache[cache_key] = response
            
            # Update usage statistics
            self._update_usage_stats(model_name, response)
            
            return response
            
        except Exception as e:
            self.logger.error(f"Error generating response with {model_name}: {e}")
            raise

    async def dual_llm_evaluation(self, prompt: str, primary_model: str, 
                                secondary_model: str, task_type: TaskType) -> Dict[str, Any]:
        """Enhanced dual-LLM evaluation with conflict resolution"""
        
        # Generate responses from both models concurrently
        primary_task = self.generate_response(prompt, primary_model, task_type)
        secondary_task = self.generate_response(prompt, secondary_model, task_type)
        
        primary_response, secondary_response = await asyncio.gather(
            primary_task, secondary_task, return_exceptions=True
        )
        
        # Handle exceptions
        if isinstance(primary_response, Exception):
            self.logger.error(f"Primary model {primary_model} failed: {primary_response}")
            return {"status": "error", "error": str(primary_response)}
        
        if isinstance(secondary_response, Exception):
            self.logger.error(f"Secondary model {secondary_model} failed: {secondary_response}")
            return {"status": "partial", "primary_response": primary_response}
        
        # Analyze agreement
        agreement_score = self._calculate_agreement(primary_response, secondary_response)
        
        result = {
            "status": "success",
            "primary_response": primary_response,
            "secondary_response": secondary_response,
            "agreement_score": agreement_score,
            "requires_human_review": agreement_score < 0.8,
            "confidence": min(primary_response.confidence, secondary_response.confidence),
            "combined_quality": (primary_response.quality_score + secondary_response.quality_score) / 2
        }
        
        # Conflict resolution
        if agreement_score < 0.8:
            result["conflict_resolution"] = await self._resolve_conflict(
                primary_response, secondary_response, prompt, task_type
            )
        
        return result

    async def batch_process(self, prompts: List[str], model_name: str, 
                          task_type: TaskType, batch_size: int = None) -> List[LLMResponse]:
        """Optimized batch processing with rate limiting and error handling"""
        
        config = self.model_configs.get(model_name)
        if not config:
            raise ValueError(f"Unknown model: {model_name}")
        
        batch_size = batch_size or config.batch_size
        results = []
        
        # Process in batches
        for i in range(0, len(prompts), batch_size):
            batch = prompts[i:i + batch_size]
            
            # Create tasks for concurrent processing
            tasks = [
                self.generate_response(prompt, model_name, task_type)
                for prompt in batch
            ]
            
            # Execute batch with rate limiting
            batch_results = await asyncio.gather(*tasks, return_exceptions=True)
            
            # Handle results and exceptions
            for j, result in enumerate(batch_results):
                if isinstance(result, Exception):
                    self.logger.error(f"Batch item {i+j} failed: {result}")
                    # Create error response
                    error_response = LLMResponse(
                        content=f"Error: {str(result)}",
                        provider=config.provider,
                        model=model_name,
                        tokens_used=0,
                        cost=0.0,
                        response_time=0.0,
                        confidence=0.0,
                        quality_score=0.0
                    )
                    results.append(error_response)
                else:
                    results.append(result)
            
            # Rate limiting delay
            if i + batch_size < len(prompts):
                delay = 60.0 / config.rate_limit_rpm * batch_size
                await asyncio.sleep(delay)
        
        return results

    def get_performance_analytics(self) -> Dict[str, Any]:
        """Comprehensive performance analytics and insights"""
        
        analytics = {
            "model_performance": {},
            "cost_analysis": {},
            "quality_metrics": {},
            "usage_patterns": {},
            "recommendations": []
        }
        
        # Model performance analysis
        for model_name, stats in self.usage_stats.items():
            if stats["total_requests"] > 0:
                analytics["model_performance"][model_name] = {
                    "total_requests": stats["total_requests"],
                    "average_response_time": stats["total_response_time"] / stats["total_requests"],
                    "average_confidence": stats["total_confidence"] / stats["total_requests"],
                    "success_rate": stats["successful_requests"] / stats["total_requests"],
                    "total_cost": stats["total_cost"],
                    "cost_per_request": stats["total_cost"] / stats["total_requests"]
                }
        
        # Generate recommendations
        analytics["recommendations"] = self._generate_recommendations()
        
        return analytics

    def optimize_model_selection(self, historical_data: List[Dict]) -> Dict[str, str]:
        """AI-driven model optimization based on historical performance"""
        
        task_recommendations = {}
        
        for task_type in TaskType:
            # Analyze historical performance for this task type
            task_data = [d for d in historical_data if d.get("task_type") == task_type.value]
            
            if not task_data:
                # Use default optimal model
                task_recommendations[task_type.value] = self.get_optimal_model(task_type)
                continue
            
            # Calculate performance scores for each model
            model_scores = {}
            for data in task_data:
                model = data["model"]
                score = (
                    data.get("quality_score", 0.5) * 0.4 +
                    (1 - data.get("response_time", 10) / 10) * 0.3 +
                    data.get("confidence", 0.5) * 0.2 +
                    (1 - data.get("cost", 1) / 1) * 0.1
                )
                
                if model not in model_scores:
                    model_scores[model] = []
                model_scores[model].append(score)
            
            # Find best performing model
            best_model = None
            best_score = 0
            
            for model, scores in model_scores.items():
                avg_score = statistics.mean(scores)
                if avg_score > best_score:
                    best_score = avg_score
                    best_model = model
            
            task_recommendations[task_type.value] = best_model or self.get_optimal_model(task_type)
        
        return task_recommendations

    # Helper methods
    def _generate_cache_key(self, prompt: str, model: str, kwargs: Dict) -> str:
        """Generate cache key for response caching"""
        content = f"{prompt}:{model}:{json.dumps(kwargs, sort_keys=True)}"
        return hashlib.md5(content.encode()).hexdigest()

    def _calculate_agreement(self, response1: LLMResponse, response2: LLMResponse) -> float:
        """Calculate agreement score between two responses"""
        # Simplified agreement calculation
        # In practice, this would use more sophisticated NLP techniques
        
        content1 = response1.content.lower().strip()
        content2 = response2.content.lower().strip()
        
        # Basic similarity check
        if content1 == content2:
            return 1.0
        
        # Check for similar decisions (include/exclude)
        if ("include" in content1 and "include" in content2) or \
           ("exclude" in content1 and "exclude" in content2):
            return 0.8
        
        # Check confidence alignment
        conf_diff = abs(response1.confidence - response2.confidence)
        if conf_diff < 0.2:
            return 0.6
        
        return 0.3

    async def _resolve_conflict(self, response1: LLMResponse, response2: LLMResponse, 
                              prompt: str, task_type: TaskType) -> Dict[str, Any]:
        """Resolve conflicts between dual LLM responses"""
        
        # Use a third, high-quality model for conflict resolution
        resolver_model = self.get_optimal_model(TaskType.CONFLICT_RESOLUTION, quality_threshold=0.9)
        
        conflict_prompt = f"""
        Two AI models provided different responses to the same systematic review task.
        
        Original Task: {prompt}
        
        Response 1: {response1.content}
        Confidence: {response1.confidence}
        
        Response 2: {response2.content}
        Confidence: {response2.confidence}
        
        Please analyze both responses and provide:
        1. Which response is more accurate and why
        2. A final decision with justification
        3. Confidence level (0-1)
        
        Provide your analysis in JSON format.
        """
        
        try:
            resolution = await self.generate_response(conflict_prompt, resolver_model, task_type)
            return {
                "resolver_model": resolver_model,
                "resolution": resolution.content,
                "resolver_confidence": resolution.confidence
            }
        except Exception as e:
            return {
                "error": f"Conflict resolution failed: {str(e)}",
                "fallback": "Human review required"
            }

    def _update_usage_stats(self, model_name: str, response: LLMResponse):
        """Update usage statistics for performance tracking"""
        
        if model_name not in self.usage_stats:
            self.usage_stats[model_name] = {
                "total_requests": 0,
                "successful_requests": 0,
                "total_response_time": 0.0,
                "total_confidence": 0.0,
                "total_cost": 0.0
            }
        
        stats = self.usage_stats[model_name]
        stats["total_requests"] += 1
        stats["successful_requests"] += 1
        stats["total_response_time"] += response.response_time
        stats["total_confidence"] += response.confidence
        stats["total_cost"] += response.cost

    def _generate_recommendations(self) -> List[str]:
        """Generate optimization recommendations based on usage patterns"""
        
        recommendations = []
        
        # Analyze cost efficiency
        high_cost_models = [
            model for model, stats in self.usage_stats.items()
            if stats.get("total_cost", 0) > 10.0 and stats.get("total_requests", 0) > 100
        ]
        
        if high_cost_models:
            recommendations.append(
                f"Consider using more cost-effective models for high-volume tasks. "
                f"High-cost models detected: {', '.join(high_cost_models)}"
            )
        
        # Analyze performance issues
        slow_models = [
            model for model, stats in self.usage_stats.items()
            if stats.get("total_requests", 0) > 0 and 
            (stats["total_response_time"] / stats["total_requests"]) > 10.0
        ]
        
        if slow_models:
            recommendations.append(
                f"Performance optimization needed for: {', '.join(slow_models)}. "
                f"Consider using faster models or implementing caching."
            )
        
        return recommendations
