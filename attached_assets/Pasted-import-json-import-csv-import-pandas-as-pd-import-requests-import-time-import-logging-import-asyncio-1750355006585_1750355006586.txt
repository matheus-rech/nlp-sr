@dataclass
class SystemConfig:
    """Configuration for the otto-SR system"""
    # LLM Configuration
    openai_api_key: Optional[str] = None
    google_api_key: Optional[str] = None
    ollama_base_url: str = "http://localhost:11434"
    
    # Model Configuration

    pdf_model: str = "gemini-2.0-flash"

    retry_attempts: int = 3
    timeout_seconds: int = 30

    confidence_threshold: float = 0.7
    require_dual_screening: bool = False

    output_format: str = "excel"  # csv, excel, json
    save_intermediate_results: bool = True
    detailed_logging: bool = True

@dataclass
class Citation:
    """Enhanced citation with processing metadata"""
    id: str
    title: str
    abstract: str
    authors: str = ""
    journal: str = ""
    year: str = ""
    doi: str = ""
    keywords: str = ""
    full_text: str = ""
    pdf_path: Optional[str] = None
    
    # Processing metadata
    processing_status: TaskStatus = TaskStatus.PENDING
    screening_result: Optional[Dict] = None
    extraction_result: Optional[Dict] = None

        result = asdict(self)

class LLMProvider:
    """Enhanced LLM provider with multiple model support"""
    
    def __init__(self, config: SystemConfig):
        self.config = config
        self.session = requests.Session()
    
    async def generate_response_async(self, prompt: str, model: str, temperature: float = 0.7, max_tokens: int = 4000) -> str:
        """Async response generation"""
        if model.startswith("gpt"):
            return await self._openai_request_async(prompt, model, temperature, max_tokens)
        elif model.startswith("gemini"):
            return await self._google_request_async(prompt, model, temperature, max_tokens)
        else:
            return await self._ollama_request_async(prompt, model, temperature, max_tokens)
    
    def generate_response(self, prompt: str, model: str, temperature: float = 0.7, max_tokens: int = 4000) -> str:
        """Sync response generation"""
        if model.startswith("gpt"):
            return self._openai_request(prompt, model, temperature, max_tokens)
        elif model.startswith("gemini"):
            return self._google_request(prompt, model, temperature, max_tokens)
        else:
            return self._ollama_request(prompt, model, temperature, max_tokens)
    
    def _openai_request(self, prompt: str, model: str, temperature: float, max_tokens: int) -> str:
        """OpenAI API request"""
        if not self.config.openai_api_key:
            return self._generate_mock_response(prompt, model)
        
        headers = {
            "Authorization": f"Bearer {self.config.openai_api_key}",
            "Content-Type": "application/json"
        }
        
        data = {
            "model": model,
            "messages": [{"role": "user", "content": prompt}],
            "temperature": temperature,
            "max_tokens": max_tokens
        }
        
        try:
            response = self.session.post(
                "https://api.openai.com/v1/chat/completions",
                headers=headers,
                json=data,
                timeout=self.config.timeout_seconds
            )
            response.raise_for_status()
            return response.json()["choices"][0]["message"]["content"]
        except Exception as e:
            logger.error(f"OpenAI API error: {e}")
            return self._generate_mock_response(prompt, model)
    
    def _google_request(self, prompt: str, model: str, temperature: float, max_tokens: int) -> str:
        """Google Gemini API request"""
        if not self.config.google_api_key:
            return self._generate_mock_response(prompt, model)
        
        # Placeholder for Google Gemini API
        # In real implementation, would use Google's API
        return self._generate_mock_response(prompt, model)
    
    def _ollama_request(self, prompt: str, model: str, temperature: float, max_tokens: int) -> str:
        """Ollama local LLM request"""
        data = {
            "model": model,
            "prompt": prompt,
            "stream": False,
            "options": {
                "temperature": temperature,
                "num_predict": max_tokens
            }
        }
        
        try:
            response = self.session.post(
                f"{self.config.ollama_base_url}/api/generate",
                json=data,
                timeout=self.config.timeout_seconds
            )
            response.raise_for_status()
            return response.json()["response"]
        except Exception as e:
            logger.error(f"Ollama API error: {e}")
            return self._generate_mock_response(prompt, model)
    
    async def _openai_request_async(self, prompt: str, model: str, temperature: float, max_tokens: int) -> str:
        """Async OpenAI API request"""
        if not self.config.openai_api_key:
            return self._generate_mock_response(prompt, model)
        
        headers = {
            "Authorization": f"Bearer {self.config.openai_api_key}",
            "Content-Type": "application/json"
        }
        
        data = {
            "model": model,
            "messages": [{"role": "user", "content": prompt}],
            "temperature": temperature,
            "max_tokens": max_tokens
        }
        
        try:
            async with aiohttp.ClientSession() as session:
                async with session.post(
                    "https://api.openai.com/v1/chat/completions",
                    headers=headers,
                    json=data,
                    timeout=aiohttp.ClientTimeout(total=self.config.timeout_seconds)
                ) as response:
                    response.raise_for_status()
                    result = await response.json()
                    return result["choices"][0]["message"]["content"]
        except Exception as e:
            logger.error(f"Async OpenAI API error: {e}")
            return self._generate_mock_response(prompt, model)
    
    async def _google_request_async(self, prompt: str, model: str, temperature: float, max_tokens: int) -> str:
        """Async Google Gemini API request"""
        return self._generate_mock_response(prompt, model)
    
    async def _ollama_request_async(self, prompt: str, model: str, temperature: float, max_tokens: int) -> str:

class ScreeningAgent:
    """Specialized agent for citation screening using GPT-4.1"""
    
    def __init__(self, llm_provider: LLMProvider, config: SystemConfig):
        self.llm_provider = llm_provider
        self.config = config
        self.agent_type = AgentType.SCREENING
    
    async def screen_citation_async(self, citation: Citation, criteria: ScreeningCriteria) -> AgentResult:
        """Async citation screening"""
        start_time = time.time()
        
        try:
            # Generate screening prompt
            prompt = self._generate_screening_prompt(citation, criteria)
            
            # Get LLM response
            response = await self.llm_provider.generate_response_async(
                prompt, 
                self.config.screening_model, 
                temperature=1.0,  # As specified in paper
                max_tokens=2000
            )
            
            # Parse response
            result_data = self._parse_screening_response(response)
            
            # Determine status based on confidence
            confidence = result_data.get('confidence', 0.0)
            if confidence >= self.config.confidence_threshold:
                status = TaskStatus.COMPLETED
            else:
                status = TaskStatus.REQUIRES_REVIEW
            
            return AgentResult(
                agent_type=self.agent_type,
                citation_id=citation.id,
                status=status,
                result_data=result_data,
                confidence=confidence,
                processing_time=time.time() - start_time,
                timestamp=datetime.now().isoformat(),
                model_used=self.config.screening_model
            )
            
        except Exception as e:
            logger.error(f"Screening error for citation {citation.id}: {e}")
            return AgentResult(
                agent_type=self.agent_type,
                citation_id=citation.id,
                status=TaskStatus.FAILED,
                result_data={},
                confidence=0.0,
                processing_time=time.time() - start_time,
                timestamp=datetime.now().isoformat(),
                model_used=self.config.screening_model,
                error_message=str(e)
            )

    def _parse_screening_response(self, response: str) -> Dict:

                "extraction_confidence": 0.0,
                "extraction_notes": f"Error parsing response: {str(e)}"
            }
    
    def _calculate_extraction_confidence(self, result_data: Dict) -> float:
        """Calculate confidence score for extraction quality"""
        base_confidence = result_data.get('extraction_confidence', 0.5)
        
        # Adjust based on completeness
        required_fields = ['study_characteristics', 'population', 'intervention', 'outcomes']
        present_fields = sum(1 for field in required_fields if field in result_data and result_data[field])
        completeness_score = present_fields / len(required_fields)
        
        # Adjust based on evidence quotes
        evidence_score = 1.0 if 'evidence_quotes' in result_data and result_data['evidence_quotes'] else 0.5
        
        # Combined confidence
        final_confidence = (base_confidence * 0.6) + (completeness_score * 0.3) + (evidence_score * 0.1)
        
        return min(final_confidence, 1.0)

class OttoSRCoordinator:
    """Main coordinator for the otto-SR multi-agent system"""
    
    def __init__(self, config: SystemConfig):
        self.config = config
        self.llm_provider = LLMProvider(config)
        
        # Initialize agents
        self.screening_agent = ScreeningAgent(self.llm_provider, config)
        self.extraction_agent = ExtractionAgent(self.llm_provider, config)
        self.pdf_processor = PDFProcessor(self.llm_provider, config)
        
        # Results storage
        self.screening_results: List[AgentResult] = []
        self.extraction_results: List[AgentResult] = []
        self.processing_log: List[Dict] = []
    
    async def process_systematic_review_async(self, 
                                            citations: List[Citation], 
                                            criteria: ScreeningCriteria, 
                                            variables: ExtractionVariables) -> Dict:
        """Complete async systematic review processing"""
        start_time = time.time()
        
        logger.info(f"Starting otto-SR processing of {len(citations)} citations")
        
        # Phase 1: Screening
        logger.info("Phase 1: Citation Screening")
        screening_tasks = []
        
        semaphore = asyncio.Semaphore(self.config.max_concurrent_tasks)
        
        async def screen_with_semaphore(citation):
            async with semaphore:
                return await self.screening_agent.screen_citation_async(citation, criteria)
        
        screening_tasks = [screen_with_semaphore(citation) for citation in citations]
        screening_results = await asyncio.gather(*screening_tasks, return_exceptions=True)
        
        # Process screening results
        included_citations = []
        for i, result in enumerate(screening_results):
            if isinstance(result, Exception):
                logger.error(f"Screening failed for citation {citations[i].id}: {result}")
                continue
            
            self.screening_results.append(result)
            
            if result.result_data.get('decision') == 'include':
                included_citations.append(citations[i])
        
        logger.info(f"Screening completed: {len(included_citations)} citations included")
        
        # Phase 2: Data Extraction
        logger.info("Phase 2: Data Extraction")
        extraction_tasks = []
        
        async def extract_with_semaphore(citation):
            async with semaphore:
                return await self.extraction_agent.extract_data_async(citation, variables)
        
        extraction_tasks = [extract_with_semaphore(citation) for citation in included_citations]
        extraction_results = await asyncio.gather(*extraction_tasks, return_exceptions=True)
        
        # Process extraction results
        for i, result in enumerate(extraction_results):
            if isinstance(result, Exception):
                logger.error(f"Extraction failed for citation {included_citations[i].id}: {result}")
                continue
            
            self.extraction_results.append(result)
        
        logger.info(f"Extraction completed: {len(self.extraction_results)} studies processed")
        
        # Generate summary
        total_time = time.time() - start_time
        summary = self._generate_processing_summary(len(citations), total_time)
        
        logger.info(f"Otto-SR processing completed in {total_time:.2f} seconds")
        
        return summary
    
    def process_systematic_review(self, 
                                 citations: List[Citation], 
                                 criteria: ScreeningCriteria, 
                                 variables: ExtractionVariables) -> Dict:
        """Complete sync systematic review processing"""
        start_time = time.time()
        
        logger.info(f"Starting otto-SR processing of {len(citations)} citations")
        
        # Phase 1: Screening
        logger.info("Phase 1: Citation Screening")
        included_citations = []
        
        with ThreadPoolExecutor(max_workers=self.config.max_concurrent_tasks) as executor:
            screening_futures = {
                executor.submit(self.screening_agent.screen_citation, citation, criteria): citation 
                for citation in citations
            }
            
            for future in as_completed(screening_futures):
                citation = screening_futures[future]
                try:
                    result = future.result()
                    self.screening_results.append(result)
                    
                    if result.result_data.get('decision') == 'include':
                        included_citations.append(citation)
                        
                except Exception as e:
                    logger.error(f"Screening failed for citation {citation.id}: {e}")
        
        logger.info(f"Screening completed: {len(included_citations)} citations included")
        
        # Phase 2: Data Extraction
        logger.info("Phase 2: Data Extraction")
        
        with ThreadPoolExecutor(max_workers=self.config.max_concurrent_tasks) as executor:
            extraction_futures = {
                executor.submit(self.extraction_agent.extract_data, citation, variables): citation 
                for citation in included_citations
            }
            
            for future in as_completed(extraction_futures):
                citation = extraction_futures[future]
                try:
                    result = future.result()
                    self.extraction_results.append(result)
                except Exception as e:
                    logger.error(f"Extraction failed for citation {citation.id}: {e}")
        
        logger.info(f"Extraction completed: {len(self.extraction_results)} studies processed")
        
        # Generate summary
        total_time = time.time() - start_time
        summary = self._generate_processing_summary(len(citations), total_time)
        
        logger.info(f"Otto-SR processing completed in {total_time:.2f} seconds")
        
        return summary
    
    def _generate_processing_summary(self, total_citations: int, processing_time: float) -> Dict:
        """Generate comprehensive processing summary"""
        included_count = sum(1 for r in self.screening_results if r.result_data.get('decision') == 'include')
        excluded_count = sum(1 for r in self.screening_results if r.result_data.get('decision') == 'exclude')
        uncertain_count = sum(1 for r in self.screening_results if r.result_data.get('decision') == 'uncertain')
        
        avg_screening_confidence = sum(r.confidence for r in self.screening_results) / len(self.screening_results) if self.screening_results else 0
        avg_extraction_confidence = sum(r.confidence for r in self.extraction_results) / len(self.extraction_results) if self.extraction_results else 0
        
        return {
            "processing_summary": {
                "total_citations": total_citations,
                "screening_results": {
                    "included": included_count,
                    "excluded": excluded_count,
                    "uncertain": uncertain_count,
                    "average_confidence": avg_screening_confidence
                },
                "extraction_results": {
                    "studies_processed": len(self.extraction_results),
                    "average_confidence": avg_extraction_confidence
                },
                "performance_metrics": {
                    "total_processing_time": processing_time,
                    "average_time_per_citation": processing_time / total_citations if total_citations > 0 else 0,
                    "citations_per_second": total_citations / processing_time if processing_time > 0 else 0
                }
            },
            "screening_results": [r.to_dict() for r in self.screening_results],
            "extraction_results": [r.to_dict() for r in self.extraction_results]
        }
    
    def save_results(self, output_path: str):
        """Save all results to files"""
        summary = self._generate_processing_summary(
            len(self.screening_results), 
            sum(r.processing_time for r in self.screening_results + self.extraction_results)
        )
        
        # Save summary
        with open(f"{output_path}_summary.json", 'w') as f:
            json.dump(summary, f, indent=2)
        
        # Save detailed results
        if self.config.output_format == 'excel':
            self._save_to_excel(output_path)
        elif self.config.output_format == 'csv':
            self._save_to_csv(output_path)
        else:
            self._save_to_json(output_path)
    
    def _save_to_excel(self, output_path: str):
        """Save results to Excel with multiple sheets"""
        with pd.ExcelWriter(f"{output_path}.xlsx", engine='openpyxl') as writer:
            # Screening results
            if self.screening_results:
                screening_df = pd.DataFrame([r.to_dict() for r in self.screening_results])
                screening_df.to_excel(writer, sheet_name='Screening_Results', index=False)
            
            # Extraction results
            if self.extraction_results:
                extraction_df = pd.DataFrame([r.to_dict() for r in self.extraction_results])
                extraction_df.to_excel(writer, sheet_name='Extraction_Results', index=False)
    
    def _save_to_csv(self, output_path: str):
        """Save results to CSV files"""
        if self.screening_results:
            screening_df = pd.DataFrame([r.to_dict() for r in self.screening_results])
            screening_df.to_csv(f"{output_path}_screening.csv", index=False)
        
        if self.extraction_results:
            extraction_df = pd.DataFrame([r.to_dict() for r in self.extraction_results])
            extraction_df.to_csv(f"{output_path}_extraction.csv", index=False)
    
    def _save_to_json(self, output_path: str):
        """Save results to JSON files"""
        if self.screening_results:
            with open(f"{output_path}_screening.json", 'w') as f:
                json.dump([r.to_dict() for r in self.screening_results], f, indent=2)
        
        if self.extraction_results:
            with open(f"{output_path}_extraction.json", 'w') as f:
                json.dump([r.to_dict() for r in self.extraction_results], f, indent=2)
